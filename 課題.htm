import json

import os

import boto3

 

REGION = os.environ['REGION']

TABLE = os.environ['TABLE']

 

dynamo = boto3.resource('dynamodb', region_name=REGION)

table = dynamo.Table(TABLE)

 

 

def lambda_handler(event, context):

 

    if event['httpMethod'] == 'GET':

        data = table.scan()['Items']

        print(data)

    elif event['httpMethod'] == 'POST':

        postData = json.loads(event['body'])

        table.put_item(Item = postData)

        data = {'message': 'successfull'}

   

    return {

        'statusCode': 200,

        'body': json.dumps(data)

    }

ロール

 

 

import json

import os

import boto3

import uuid

 

REGION = os.environ['REGION']

TABLE = os.environ['TABLE']

 

dynamo = boto3.resource('dynamodb', region_name=REGION)

table = dynamo.Table(TABLE)

 

 

def lambda_handler(event, context):

 

    if event['httpMethod'] == 'GET':

        data = table.scan()['Items']

        print(data)

    elif event['httpMethod'] == 'POST':

        postData = json.loads(event['body'])

        postData['id'] = str(uuid.uuid1())

        table.put_item(Item = postData)

        data = {'message': 'successfull'}

   

    return {

        'statusCode': 200,

        'headers': {

            'Access-Control-Allow-Headers': 'Content-Type',

            'Access-Control-Allow-Origin': "https://d32syb6js4m9ls.cloudfront.net/",

            'Access-Control-Allow-Methods': 'GET',

        },

        'body': json.dumps(data)

    }

 

 

・boto3　awsのサービスにアクセスするためのもの

・lambda_handler はlambda関数の際に実行される

・キーを指定して欲しいときはget_item

・全件欲しいときはtable.scan()

・Clientから渡されるデータはeventに入る

・値を確認したい場合はprint(確認したい値)　→　実行　→　CloudFrontで確認できる

・アプリからどの値が来ているのかを確認する print(event['body'])

・json.loads(event[value']) データを辞書型に変更する

・アプリから送られてきた値をテーブルに格納したい場 table.put_item(Item=json.loads(event['body]'))

・ロールにはputと

dynamo = boto3.resource('dynamodb', region_name=REGION)

def lambda_handler(event, context):

   

    response = table.scan()

    item = response['Items']

    return {

        'statusCode': 200,
